#### Q1.3

```
(define
    (square-two-of-max x y z)
    (if (< x y)
      (if (< x z)
          (+ (* y y) (* z z))
          (+ (* x x) (* y y)))
      (if (< y z)
          (+ (* x x) (* z z))
          (+ (* x x) (* y y)))))
```

```
(define
    (square-two-of-max x y z)
    (cond ((and (< x y) (< x z)) (+ (* y y) (* z z)))
          ((and (< y x) (< y z)) (+ (* x x) (* z z)))
          ((and (< z x) (< z y)) (+ (* x x) (* y y)))))
```
